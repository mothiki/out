/**
 * Pads the str String with 0s and returns the modified String
 * @param {String} str String to be padded 
 * @param {int} max The total length required in the returned string
 * @returns {String} A 0-prefixed str, such that the length of str is max
 */   
function pad (str, max) {
    str = str.toString();
    return str.length < max ? pad("0" + str, max) : str;
}

/**
 * Returns DateTime in 24 Hour format
 * @param {Date} dateObj A DateTime Object
 * @returns {String} A String representing the dateObj in 24 hour time format 
 */
function get24HourDateTime(dateObj){
	return (dateObj.toLocaleDateString() + " " + dateObj.getHours() + ":" + dateObj.getMinutes() + ":" + dateObj.getSeconds())
}

/**
 * Checks if is a number
 * @param {any} n Parameter to be checked if numeric
 * @returns {boolean} returns True if n is numeric, and False otherwise
 */
function isNumeric(n){
	return !isNaN(parseFloat(n)) && isFinite(n);
}

/**
 * Checks if is a positive number or 0
 * @param {any} n Parameter to be checked
 * @returns {boolean} returns True if n is 0 or a positive number, and False otherwise
 * 
 */
function isPositiveNumber(n){
	if(isNumeric(n)){
		if(parseFloat(n) >= 0){
			return true;
		}
	}
	return false;
}
