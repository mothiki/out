/**
 * Calibrates the Camera and Saves the calibration information of the camera connected with the local spectrometer 
 * 
 */
function cameraCalibrate(){
	alert("Starting Camera Calibration")
	alert("Put Bright Screen")
	var brightObjPath, darkObjPath;
	
	 $.ajax({
		    type: "POST",
		    url: '/calibrateCameraCapture',
		    cache: false,
			data: {'colorFlag': 'bright'},
		    success: function (brightData) {
		    	console.log(brightData);
		        brightObjPath = brightData;
		        alert("Put dark screen")
		    	$.ajax({
				    type: "POST",
				    url: '/calibrateCameraCapture',
				    cache: false,
					data: {'colorFlag': 'dark'},
				    success: function (darkData) {
				    	console.log(darkData);
			        	darkObjPath = darkData;
			            alert("Put dark screen");
			            var confirmed = confirm("Do you want to save this Calibration?");
			            if(confirmed){
			            	$.ajax({
							    type: "POST",
							    url: '/calibrateCameraSave',
							    cache: false,
								data: {
									'brightColorObjPath' : brightObjPath,
									'darkColorObjPath' : darkObjPath
								},
							    success: function (data) {
							    	console.log(data);
				        	        
							    },
							    error: function (msg) {
							    	console.log(msg)
							    }
							 })
			            }
				    	
				    },
				    error: function (msg) {
				    	console.log(msg)
				    }
				 })
		    },
		    error: function (msg) {
		    	console.log(msg)
		    }
		 });
	
	
}
